#include "Patient.h"



Patient::Patient()
{
}
Patient::Patient(string n, string mn, string ln, string a, string c, string s, string z, string p, string ename, string enumber) {
	name = n;
	middleName = mn;
	lastName = ln;
	address = a;
	city = c;
	state = s;
	zip = z;
	phoneNumber = p;
	emergencyName = ename;
	emergencyNumber = enumber;
}

Patient::~Patient()
{
}
void Patient::setName(string n) {
	name = n;
}
string Patient::getName() {
	return name;
}
void Patient::setMiddleName(string n) {
	middleName = n;
}
string Patient::getMiddleName() {
	return middleName;
}
void Patient::setLastName(string n) {
	lastName = n;
}
string Patient::getLastName() {
	return lastName;
}
void Patient::setAddress(string n) {
	address = n;
}
string Patient::getAddress() {
	return address;
}
void Patient::setCity(string n) {
	city = n;
}
string Patient::getCity() {
	return city;
}
void Patient::setState(string n) {
	state = n;
}
string Patient::getState() {
	return state;
}
void Patient::setZip(string n) {
	zip = n;
}
string Patient::getZip() {
	return zip;
}
void Patient::setPhoneNumber(string n) {
	phoneNumber = n;
}
string Patient::getPhoneNumber() {
	return phoneNumber;
}
void Patient::setEmergencyName(string n) {
	emergencyName = n;
}
string Patient::getEmergencyName() {
	return emergencyName;
}
void Patient::setEmergencyNumber(string n) {
	emergencyNumber = n;
}
string Patient::getEmergencyNumber() {
	return emergencyNumber;
}
